from langgraph.graph import StateGraph, END

# 1. Define the Graph
workflow = StateGraph(AgentState)

# 2. Add Nodes (The boxes in the diagram)
workflow.add_node("researcher", research_func)
workflow.add_node("writer", writer_func)

# 3. Add Normal Edges (The straight lines)
workflow.set_entry_point("researcher")
workflow.add_edge("researcher", "writer")

# 4. Add the Loop (The curved arrow)
workflow.add_conditional_edges(
    "writer",
    should_continue, # Decides where to go next
    {
        "improve": "researcher", # Loop back
        "finish": END            # Exit graph
    }
)
